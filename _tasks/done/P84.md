### Key Pattern
```go
// Core handler (thin wrapper)
func (s *BaseServer) handleContainerCreate(w http.ResponseWriter, r *http.Request) {
    var req api.ContainerCreateRequest
    ReadJSON(r, &req)
    resp, err := s.self.ContainerCreate(&req)  // virtual dispatch
    WriteJSON(w, http.StatusCreated, resp)
}

// Cloud backend override (typed method)
```
